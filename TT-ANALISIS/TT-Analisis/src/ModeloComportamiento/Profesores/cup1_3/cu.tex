\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar una academia dando clic en el icono %\IUEliminar 
	de la pantalla \cdtIdRef{IUP1}{Gestionar Profesores}.
	
	\UCpaso [\UCsist] Verifica que el profesor no haya sido asociado a una o más academias y/o Unidades de Aprendizaje. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar el profesor oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina el profesor.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUP1}{Gestionar Profesores} indicando que el profesor se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El profesor ya fue asociado al menos a una academia o con una Unidad de Aprendizaje.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUP1}{Gestionar Profesores}. 
\end{UCtrayectoriaA}
