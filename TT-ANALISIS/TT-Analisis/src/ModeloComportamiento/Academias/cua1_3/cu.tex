\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar una academia dando clic en el icono %\IUEliminar 
	de la pantalla \cdtIdRef{IUA1}{Gestionar Academias}
	
	\UCpaso [\UCsist] Verifica que la academia no haya sido asociada a una o más Unidades de Aprendizaje. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar la academia oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina la academia.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUA1}{Gestionar Academias} indicando que la academia se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{La academia ya fue asociada a una Unidad de Aprendizaje.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUA1}{Gestionar Academias}. 
\end{UCtrayectoriaA}
