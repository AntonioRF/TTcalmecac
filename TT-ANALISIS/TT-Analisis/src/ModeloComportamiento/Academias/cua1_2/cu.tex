\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita modificar una academia dando clic en el ícono %\IUModificar
	de la pantalla \cdtIdRef{IUA1}{Gestionar Academias}. 
	
	\UCpaso [\UCsist] Verifica que exista al menos un profesor registrado. \refTray{A}
	
	\UCpaso [\UCsist] Obtiene el nombre completo de los profesores basificados con base en la regla de negocios \cdtIdRef{RN-N4}{Jefe de academia}.
	
	\UCpaso [\UCsist] Obtiene la información de la academia previamente registrada.

	\UCpaso[\UCsist] Solicita la información a través de la pantalla \cdtIdRef{IUA1.1}{Registrar Academia} con la información obtenida.
	
	\UCpaso [\UCactor] Completa la información solicitada. \label{CUA1.1:CompletaInfo}
	
	\UCpaso [\UCactor] Solicita modificar la información oprimiendo el botón \cdtButton{Aceptar}. \refTray{B}
	
	\UCpaso [\UCsist] Verifica que se hayan completado los campos obligatorios con base en la regla de negocios \cdtIdRef{RN-S1}{Datos requeridos}. \refTray{C}
	
	\UCpaso [\UCsist] Verifica que el nombre de la academia no se haya registrado previamente, con base en la regla de negocios \cdtIdRef{RN-S2}{Unicidad de elementos}. \refTray{D}
		
	\UCpaso [\UCsist] Verifica que el profesor seleccionado como jefe de academia no sea ya jefe de otra academia, con base en la regla de negocios \cdtIdRef{RN-S2}{Unicidad de elementos}. \refTray{E}
	
	\UCpaso [\UCsist] Modifica el nombre y jefe de la academia.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUA1}{Gestionar Academias}, indicando que la academia se modificó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El catálogo de profesores no tiene información.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG7}{No existe información necesaria en el sistema} en la pantalla \cdtIdRef{IUA1}{Gestionar Academias}. 
\end{UCtrayectoriaA}

\begin{UCtrayectoriaA}[Fin del caso de uso]{B}{El actor desea cancelar la operación.}
	\UCpaso [\UCsist] Muestra la pantalla \cdtIdRef{IUA1}{Gestionar Academias}. 
\end{UCtrayectoriaA}

\begin{UCtrayectoriaA}{C}{No se completaron los campos obligatorios.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG3}{Falta dato obligatorio} en la pantalla \cdtIdRef{IUA1.1}{Registrar Academia}.
	
	\UCpaso Regresa al paso \ref{CUA1.1:CompletaInfo} de la trayectoria principal.
\end{UCtrayectoriaA}

\begin{UCtrayectoriaA}{D}{El nombre de la academia ya fue registrado.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG6}{Elemento duplicado} en la pantalla \cdtIdRef{IUA1.1}{Registrar Academia}.
	
	\UCpaso Regresa al paso \ref{CUA1.1:CompletaInfo} de la trayectoria principal.
\end{UCtrayectoriaA}

\begin{UCtrayectoriaA}{E}{El profesor seleccionado ya es jefe de otra academia.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG6}{Elemento duplicado} en la pantalla \cdtIdRef{IUA1.1}{Registrar Academia}.
	
	\UCpaso Regresa al paso \ref{CUA1.1:CompletaInfo} de la trayectoria principal.
\end{UCtrayectoriaA}
