\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Solicita gestionar los espacios oprimiendo el botón \cdtButton{Espacios} de la pantalla \cdtIdRef{IUI1}{Gestionar Edificios}.
	
	\UCpaso [\UCsist] Obtiene el nombre y número de niveles del edificio.
	
	\UCpaso [\UCsist] Obtiene la clave, nombre y capacidad de los espacios contenidos en dicho edificio y nivel.
	
	\UCpaso [\UCsist] Ordena los espacios de forma ascendente de acuerdo a su clave.
	
	\UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUI1.5}{Gestionar Espacios} con la información de los espacios obtenidos y los botones \cdtButton{Modificar}, \cdtButton{Eliminar} y \cdtButton{Consultar}, y el ícono \btnRegistrar. 
	\label{cui1.5:gestionar}
\end{UCtrayectoria}
