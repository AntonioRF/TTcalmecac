\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar un espacio dando clic en el icono IUEliminar 
	de la pantalla \cdtIdRef{IUI1.5}{Gestionar espacios}
	
	\UCpaso [\UCsist] Verifica que el espacio no haya sido asociado a uno o más grupos o estructuras académicas. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar el espacio oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina el espacio.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUI1.5}{Gestionar espacios} indicando que el espacio se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{La espacio ya fue asociada a una Unidad de Aprendizaje.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUA1}{Gestionar espacios}. 
\end{UCtrayectoriaA}
