\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar un edificio dando clic en el icono IUEliminar de la pantalla \cdtIdRef{IUI1}{Gestionar Edificios}.
	
	\UCpaso [\UCsist] Verifica que el edificio no tenga un espacio asociado. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación para eliminar el edificio a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar el espacio oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina el espacio.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUI1}{Gestionar Edificios} indicando que el edificio se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El edificio ya tiene un espacio registrado.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUI1}{Gestionar Edificios}, indicando que el edificio ya tiene un espacio registrado. 
\end{UCtrayectoriaA}
