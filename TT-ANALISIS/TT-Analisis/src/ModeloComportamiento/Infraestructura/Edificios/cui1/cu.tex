
\begin{UseCase}{CUI1}{Gestionar edificios}{
				
	Permite al actor registrar un nuevo edificio, consultar, modificar y eliminar los registros de los edificios ya existentes para la ESCOM. Esto permitirá tener un control de la infraestructura de la escuela, y posteriormente se podrán registrar los espacios existentes.
}

\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Selecciona la opción ''Edificios'' del menú \textbf{Edificios}.
	
	\UCpaso [\UCsist] Obtiene el nombre y número de niveles del edificio.
	
	\UCpaso [\UCsist] Ordena alfabéticamente los edificios por nombre.
	
	\UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUI1}{Gestionar Edificios} con la información de los edificios obtenida con los botones \cdtButton{Modificar}, \cdtButton{Eliminar}, \cdtButton{Consultar} y \cdtButton{Espacios}, y con el ícono \btnRegistrar. 
	\label{cui1:gestionar}
\end{UCtrayectoria}
