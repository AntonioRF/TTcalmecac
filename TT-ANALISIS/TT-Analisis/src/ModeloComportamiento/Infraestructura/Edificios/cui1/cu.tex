
\begin{UseCase}{CUI1}{Gestionar edificios}{	
	Permite al Subdirector Académico registrar un nuevo edificio, consultar, modificar y eliminar los registros de los edificios ya existentes para la ESCOM. Esto permitirá tener un control de la infraestructura de la escuela.
}

\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Selecciona la opción ''Edificios'' del menú \textbf{Edificios}. %\cdtIdRef{MN1}{Menú de la Secretaría de Posgrado}.
	
	\UCpaso [\UCsist] Obtiene el nombre del edificio y el número de niveles.
	
	\UCpaso [\UCsist] Ordena alfabéticamente los edificios por el nombre.
	
	\UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUI1}{Gestionar Edificios} con la información de los edificios obtenida e iconos IUModificar, IUEliminar y IURegistrar. 
	\label{cui1:gestionar}
\end{UCtrayectoria}
