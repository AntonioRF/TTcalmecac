\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita consultar la información de un profesor oprimiendo el botón \cdtButton{Consultar}
	de la pantalla \cdtIdRef{IUP1}{Gestionar Profesores}.
	
	\UCpaso [\UCsist] Obtiene el nombre, primer apellido, segundo apellido, RFC y si tiene un cargo especial el profesor seleccionado.
	
	\UCpaso [\UCsist] Muestra la pantalla \cdtIdRef{IUP1.4}{Consultar Profesor} con la información obtenida.
	
	\UCpaso [\UCactor] Solicita terminar la consulta oprimiendo el botón \cdtButton{Regresar} de la pantalla \cdtIdRef{IUP1.4}{Consultar Profesor}.
	
	\UCpaso [\UCsist] Muestra la pantalla \cdtIdRef{IUP1}{Gestionar Profesores}.
	
\end{UCtrayectoria}
