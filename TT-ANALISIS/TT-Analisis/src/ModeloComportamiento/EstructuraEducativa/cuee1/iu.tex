\subsubsection{Comandos}
    \begin{itemize}
		\item \btnRegistrar: Permite al actor registrar un nuevo profesor, dirige a la pantalla \cdtIdRef{IUP1.1}{Registrar Profesor}.
		
		\item \cdtButton{Modificar}: Permite al actor modificar el contenido de un profesor previamente registrado, dirige a la pantalla \cdtIdRef{IUP1.2}{Modificar Profesor}.
		
		\item \cdtButton{Eliminar}: Permite al actor eliminar un profesor previamente registrada, dirige a la pantalla \cdtIdRef{IUP1}{Gestionar Profesor}.
	
		\item \cdtButton{Consultar}: Permite al actor consultar la informaci칩n registrada para un profesor, dirige a la pantalla \cdtIdRef{IUP1.4}{Consultar Profesor}.
    \end{itemize}
