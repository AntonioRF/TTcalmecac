\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar un profesor oprimiendo el botón \cdtButton{Eliminar} 
	de la pantalla \cdtIdRef{IUAP1}{Gestionar Asociación de Profesores}.
	
	\UCpaso [\UCsist] Verifica que la asociación del profesor no haya sido utilizada en la generación de horarios. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar la asociación de un profesor oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina la asociación de un  profesor.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUAP1}{Gestionar Asociación de Profesores} indicando que la asociación se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{La asociación de un profesor ya ha sido utilizada en la generación de horarios.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUAP1}{Gestionar Asociación de Profesores}. 
\end{UCtrayectoriaA}
