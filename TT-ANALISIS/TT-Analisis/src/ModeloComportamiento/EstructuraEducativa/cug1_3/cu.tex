\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar un profesor oprimiendo el botón \cdtButton{Eliminar} 
	de la pantalla \cdtIdRef{IUG1}{Gestionar Grupos}.
	
	\UCpaso [\UCsist] Verifica que el grupo no haya sido asociado a un espacio. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar el grupo oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina el grupo.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUG1}{Gestionar Grupos} indicando que el grupo se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El grupo ya fue asociado con un espacio.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUG1}{Gestionar Grupos}. 
\end{UCtrayectoriaA}
