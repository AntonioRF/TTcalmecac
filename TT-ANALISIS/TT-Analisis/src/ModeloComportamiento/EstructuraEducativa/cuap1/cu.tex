
\begin{UseCase}{CUAP1}{Gestionar asociación de profesores}{	
		
	Permite al actor registrar la asociación de un profesor con una unidad de aprendizaje, consultar, modificar y eliminar estas asociaciones. Esto permitirá mantener la información actualizada y poder utilizar esta tupla en la generación de los horarios.
}

\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Solicita gestionar la asociación de profesores oprimiendo el botón \cdtButton{Asociar profesores} de la pantalla \cdtIdRef{IUEE1}{Gestionar Estructura Educativa}.
	
	\UCpaso [\UCsist] Obtiene el nombre del profesor y nombre de la unidad de aprendizaje.
	
	\UCpaso [\UCsist] Ordena de forma ascendente los registros mediante el nombre del profesor.
	
	\UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUAP1}{Gestionar Asociación de Profesores} con la información obtenida con los botones \cdtButton{Modificar}, \cdtButton{Eliminar} y \cdtButton{Consultar}, con el ícono \btnRegistrar. \label{cuap1:gestionar}
\end{UCtrayectoria}
