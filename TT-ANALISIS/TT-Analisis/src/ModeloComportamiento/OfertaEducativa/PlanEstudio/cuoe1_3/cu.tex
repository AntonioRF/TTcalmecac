	Permite al actor eliminar un plan de estudio debido a que este se registró por error o ya no se desea utilizar. Sin embargo, un plan de estudio que ya tiene asociada al menos una unidad de aprendizaje no puede ser eliminado.

\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita eliminar un plan de estudio dando clic en el icono IUEliminar de la pantalla \cdtIdRef{IUOE1}{Gestionar Plan de Estudio}.
	
	\UCpaso [\UCsist] Verifica que el plan de estudio no tenga una o más unidades de aprendizaje asociadas. \refTray{A}
	
	\UCpaso [\UCsist] Solicita la confirmación para eliminar el plan de estudio a través del mensaje \cdtIdRef{MSG2}{Eliminar Elemento}.
	
	\UCpaso [\UCactor] Confirma eliminar el plan de estudio oprimiendo el botón \cdtButton{Si}.
	
	\UCpaso [\UCsist] Elimina el plan de estudio.
	
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \cdtIdRef{IUOE1}{Gestionar Planes de Estudio} indicando que el plan de estudio se eliminó correctamente.	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El plan de estudio ya tiene una o más unidades de aprendizaje asociadas.}
	\UCpaso [\UCsist] Muestra el mensaje \cdtIdRef{MSG5}{No es posible eliminar un elemento} en la pantalla \cdtIdRef{IUOE1}{Gestionar Plan de Estudio}, indicando que el plan de estudio ya tiene una o más unidades de aprendizaje asociadas. 
\end{UCtrayectoriaA}
