\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita consultar una unidad de aprendizaje dando clic en el ícono IUConsultar de la pantalla \cdtIdRef{IUOE1.5}{Gestionar Unidades de Aprendizaje}. 
	
	\UCpaso [\UCsist] Obtiene la información de la unidad de aprendizaje seleccionada.
	
	\UCpaso [\UCsist] Muestra la pantalla \cdtIdRef{IUOE1.5.4}{Consultar Unidad de Aprendizaje} con la información obtenida.
	
	\UCpaso [\UCactor] Solicita finalizar la consulta oprimiendo el botón \cdtButton{Regresar}. 
	
	\UCpaso [\UCsist] Muestra la pantalla \cdtIdRef{IUOE1.5}{Gestionar Unidades de Aprendizaje}.
